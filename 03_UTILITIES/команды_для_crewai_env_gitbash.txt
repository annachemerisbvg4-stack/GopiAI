# Команды для создания и использования окружения crewai_env (git bash, Windows)

# 1. (Опционально) Сохранить список старых пакетов и очистить rag_memory_env
source rag_memory_env/Scripts/activate
pip freeze > requirements_old.txt
pip uninstall -y $(pip freeze | awk -F '==' '{print $1}')
deactivate

# 2. Создать новое окружение для CrewAI+LLM
python -m venv crewai_env
source GopiAI-CrewAI/crewai_env/Scripts/activate

# 3. Установить только нужные зависимости
pip install crewai 'protobuf>=6.30.0,<7.0.0'
# (Добавьте другие LLM по необходимости)

# 4. Запускать тесты CrewAI/LLM только из этого окружения
# (Перед запуском активировать)
source GopiAI-CrewAI/crewai_env/Scripts/activate
python GopiAI-CrewAI/test_crewai_groq_minimal.py
